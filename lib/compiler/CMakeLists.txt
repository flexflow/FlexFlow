     CONFIGURE_DEPENDS
     LIST_DIRECTORIES False
     src/*.cc)

  ${project_target}
  SHARED
  ${SRC}
)

  ${project_target}
  PRIVATE
    src/

    include/
)

  ${project_target}
  op-attrs
  utils
  json
  optional
  pcg
  spdlog
  # substitutions
)
